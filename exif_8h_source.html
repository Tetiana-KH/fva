41//
42// Class responsible for storing and parsing EXIF information from a JPEG blob
43//
44class EXIFInfo {
45 public:
46 // Parsing function for an entire JPEG image buffer.
47 //
48 // PARAM 'data': A pointer to a JPEG image.
49 // PARAM 'length': The length of the JPEG image.
50 // RETURN: PARSE_EXIF_SUCCESS (0) on succes with 'result' filled out
51 // error code otherwise, as defined by the PARSE_EXIF_ERROR_* macros
52 int parseFrom(const unsigned char *data, unsigned length);
53 int parseFrom(const std::string &data);
54
55 // Parsing function for an EXIF segment. This is used internally by parseFrom()
56 // but can be called for special cases where only the EXIF section is
57 // available (i.e., a blob starting with the bytes "Exif\0\0").
58 int parseFromEXIFSegment(const unsigned char *buf, unsigned len);
59
60 // Set all data members to default values.
61 void clear();
62
63 // Data fields filled out by parseFrom()
64 char ByteAlign; // 0 = Motorola byte alignment, 1 = Intel
65 std::string ImageDescription; // Image description
66 std::string Make; // Camera manufacturer's name
67 std::string Model; // Camera model
68 unsigned short Orientation; // Image orientation, start of data corresponds to
69 // 0: unspecified in EXIF data
70 // 1: upper left of image
71 // 3: lower right of image
72 // 6: upper right of image
73 // 8: lower left of image
74 // 9: undefined
75 unsigned short BitsPerSample; // Number of bits per component
76 std::string Software; // Software used
77 std::string DateTime; // File change date and time
78 std::string DateTimeOriginal; // Original file date and time (may not exist)
79 std::string DateTimeDigitized; // Digitization date and time (may not exist)
80 std::string SubSecTimeOriginal; // Sub-second time that original picture was taken
81 std::string Copyright; // File copyright information
82 double ExposureTime; // Exposure time in seconds
83 double FNumber; // F/stop
84 unsigned short ExposureProgram; // Exposure program
85 // 0: Not defined
86 // 1: Manual
87 // 2: Normal program
88 // 3: Aperture priority
89 // 4: Shutter priority
90 // 5: Creative program
91 // 6: Action program
92 // 7: Portrait mode
93 // 8: Landscape mode
94 unsigned short ISOSpeedRatings; // ISO speed
95 double ShutterSpeedValue; // Shutter speed (reciprocal of exposure time)
96 double ExposureBiasValue; // Exposure bias value in EV
97 double SubjectDistance; // Distance to focus point in meters
98 double FocalLength; // Focal length of lens in millimeters
99 unsigned short FocalLengthIn35mm; // Focal length in 35mm film
100 char Flash; // 0 = no flash, 1 = flash used
101 unsigned short FlashReturnedLight;// Flash returned light status
102 // 0: No strobe return detection function
103 // 1: Reserved
104 // 2: Strobe return light not detected
105 // 3: Strobe return light detected
106 unsigned short FlashMode; // Flash mode
107 // 0: Unknown
108 // 1: Compulsory flash firing
109 // 2: Compulsory flash suppression
110 // 3: Automatic mode
111 unsigned short MeteringMode; // Metering mode
112 // 1: average
113 // 2: center weighted average
114 // 3: spot
115 // 4: multi-spot
116 // 5: multi-segment
117 unsigned ImageWidth; // Image width reported in EXIF data
118 unsigned ImageHeight; // Image height reported in EXIF data
119 struct Geolocation_t { // GPS information embedded in file
120 double Latitude; // Image latitude expressed as decimal
121 double Longitude; // Image longitude expressed as decimal
122 double Altitude; // Altitude in meters, relative to sea level
123 char AltitudeRef; // 0 = above sea level, -1 = below sea level
124 double DOP; // GPS degree of precision (DOP)
125 struct Coord_t {
126 double degrees;
127 double minutes;
128 double seconds;
129 char direction;
130 } LatComponents, LonComponents; // Latitude, Longitude expressed in deg/min/sec
131 } GeoLocation;
132 struct LensInfo_t { // Lens information
133 double FStopMin; // Min aperture (f-stop)
134 double FStopMax; // Max aperture (f-stop)
135 double FocalLengthMin; // Min focal length (mm)
136 double FocalLengthMax; // Max focal length (mm)
137 double FocalPlaneXResolution; // Focal plane X-resolution
138 double FocalPlaneYResolution; // Focal plane Y-resolution
139 unsigned short FocalPlaneResolutionUnit; // Focal plane resolution unit
140 // 1: No absolute unit of measurement.
141 // 2: Inch.
142 // 3: Centimeter.
143 // 4: Millimeter.
144 // 5: Micrometer.
145 std::string Make; // Lens manufacturer
146 std::string Model; // Lens model
147 } LensInfo;
148
149
150 EXIFInfo() {
151 clear();
152 }
153};
