19void fvaPopulateInputDir(const QString& folder, QTreeWidgetItem* item, QTreeWidget* treeWidget, const QList<QString> & problemFileList)
20{
21 QDir dir(folder);
22 Q_FOREACH(QFileInfo info, dir.entryInfoList(QDir::NoDotAndDotDot | QDir::System | QDir::Hidden | QDir::AllDirs | QDir::Files, QDir::DirsFirst))
23 {
24 if ( !info.isDir() && problemFileList.isEmpty() )
25 continue;

29 continue;
30

32 treeWidgetItem->setText ( 0, info.fileName() );

34 treeWidgetItem->setData( 1, 1, (QString) info.absoluteFilePath() );
35
36 if (info.isDir() && info.fileName().length()!=4) // not YEAR folder
37 treeWidgetItem->setForeground( 0 , QBrush (Qt::red) );
38 if (!info.isDir()) // now it is a file
39 {
40 for (auto it = problemFileList.begin(); it != problemFileList.end(); ++it)
41 {
42 QString fileName = *it;
43 if (fileName.toUpper() == info.absoluteFilePath().toUpper())
44 treeWidgetItem->setForeground( 0 , QBrush (Qt::red) );
45 }
46 }
47 if (item)
48 {
49 item->addChild(treeWidgetItem);
50 if (info.isDir())
51 {
52 QIcon icon = QIcon(QCoreApplication::applicationDirPath() + "/Icons/folder.png");
53 item->setIcon(0, icon);
54 }
55 }
56 else
57 treeWidget->addTopLevelItem (treeWidgetItem);
58
59 fvaPopulateInputDir(info.absoluteFilePath(), treeWidgetItem, treeWidget, problemFileList);
60 }

83
84 if (!text.isEmpty())
85 {
86 // tell the painter to draw on the QImage

94
95 QPixmap _qpSource = QPixmap::fromImage(image);
96 QPixmap _qpCurrent = QPixmap::fromImage(image);
97
98 float cw = imgLabel->width(), ch = imgLabel->height();
99 float pw = _qpCurrent.width(), ph = _qpCurrent.height();
100
101 if (pw > cw && ph > ch && pw/cw > ph/ch || //both width and high are bigger, ratio at high is bigger or
102 pw > cw && ph <= ch || //only the width is bigger or
103 pw < cw && ph < ch && cw/pw < ch/ph //both width and height is smaller, ratio at width is smaller
104 )
105 _qpCurrent = _qpSource.scaledToWidth(cw, Qt::TransformationMode::FastTransformation);
106 else if (pw > cw && ph > ch && pw/cw <= ph/ch || //both width and high are bigger, ratio at width is bigger or
107 ph > ch && pw <= cw || //only the height is bigger or
108 pw < cw && ph < ch && cw/pw > ch/ph //both width and height is smaller, ratio at height is smaller
109 )
110 _qpCurrent = _qpSource.scaledToHeight(ch, Qt::TransformationMode::FastTransformation);
111
112 imgLabel->setPixmap(_qpCurrent);
113
114 return FVA_NO_ERROR;

322void fvaUpdateChecks(QTreeWidgetItem *item, int column)
323{
324 bool diff = false;
325 if(column != 0 && column!=-1)
326 return;
327 if(item->childCount()!=0 && item->checkState(0)!=Qt::PartiallyChecked && column!=-1)
328 {
329 Qt::CheckState checkState = item->checkState(0);
330 for (int i = 0; i < item->childCount(); ++i)
331 item->child(i)->setCheckState(0, checkState);
332
333 }
334 else if (item->childCount()==0 || column==-1)
335 {
336 if(item->parent()==0)
337 return;
338 for (int j = 0; j < item->parent()->childCount(); ++j)
339 {
340 if(j != item->parent()->indexOfChild(item) && item->checkState(0)!=item->parent()->child(j)->checkState(0))
341 {
342 diff = true;
343 }
344 }
345 if(diff)
346 item->parent()->setCheckState(0,Qt::PartiallyChecked);
347 else
348 item->parent()->setCheckState(0,item->checkState(0));
349 if(item->parent()!=0)
350 fvaUpdateChecks(item->parent(),-1);
351 }
