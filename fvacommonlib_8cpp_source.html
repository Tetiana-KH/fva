49FVA_EXIT_CODE fvaParseDirName(const QString& dirName, QDateTime& from, QDateTime& to, const FvaFmtContext& ctx)
50{
51 switch( dirName.length() )
52 {
53 case 4: // one year folder
54 {
55 from = QDateTime::fromString(dirName, ctx.fvaDirNameYear);
56 if ( !from.isValid() )
57 return FVA_ERROR_WRONG_FOLDER_NAME;
58 to = from/*.addYears(1)*/;
59 }
60 break;
61 case 9: // year period
62 {
63 if ( dirName[ 4 ] != '-' )
64 return FVA_ERROR_WRONG_FOLDER_NAME;
65
66 from = QDateTime::fromString( dirName.mid( 0, 4 ), ctx.fvaDirNameYear);
67 to = QDateTime::fromString(dirName.mid(5, 4), ctx.fvaDirNameYear);
68
69 if ( !from.isValid() || !to.isValid() )
70 return FVA_ERROR_WRONG_FOLDER_NAME;
71 }
72 break;
73 case 10 : // one-day event
74 {
75 from = QDateTime::fromString( dirName, ctx.fvaDirName);
76 if ( !from.isValid() )
77 return FVA_ERROR_WRONG_FOLDER_NAME;
78 to = from.addDays(1);
79 }
80 break;
81 case 13 :
82 {
83 from = QDateTime::fromString(dirName.mid( 0,10 ), ctx.fvaDirName);
84 if ( !from.isValid() )
85 return FVA_ERROR_WRONG_FOLDER_NAME;
86 if ( dirName [ 10 ] == ' ' ) // one day and several events
87 {
88 if ( dirName [ 11 ] != '#' )
89 return FVA_ERROR_WRONG_FOLDER_NAME;
90 else
91 {
92 bool result = false;
93 int dEventNumber = dirName.mid( 12, 1 ).toInt( &result );
94 if ( !result || !dEventNumber )
95 return FVA_ERROR_WRONG_FOLDER_NAME;
96 }
97 to = from.addDays(1);
98 }
99 else if ( dirName [ 10 ] == '-' ) // period
100 {
101 QString sEndDate = dirName.mid( 11,2 );
102 QString sStartDate = dirName.mid( 8,2 );
103 bool res, res1 = false;
104 int dEndDate = sEndDate.toInt( &res );
105 int dStartDate = sStartDate.toInt( &res1 );
106 if ( !res || !res1 || !dEndDate || !dStartDate)
107 return FVA_ERROR_WRONG_FOLDER_NAME;
108 to = from.addDays(dEndDate-dStartDate);
109 to = to.addDays(1);
110 }
111 else
112 return FVA_ERROR_WRONG_FOLDER_NAME;
113 }
114 break;
115 case 16: //months-day period
116 {
117 from = QDateTime::fromString(dirName.mid( 0,10 ), ctx.fvaDirName);
118 if ( !from.isValid() )
119 return FVA_ERROR_WRONG_FOLDER_NAME;
120 if ( dirName [ 10 ] != '-' ) // not a period
121 return FVA_ERROR_WRONG_FOLDER_NAME;
122
123 QString sTo = dirName.mid( 0, 4 ) + "." + dirName.mid( 11,5 );
124 to = QDateTime::fromString(sTo, ctx.fvaDirName);
125 if ( !to.isValid() )
126 return FVA_ERROR_WRONG_FOLDER_NAME;
127 to = to.addDays(1);
128 }
129 break;
130 default:
131 return FVA_ERROR_WRONG_FOLDER_NAME;
132 }
133 return FVA_NO_ERROR;
134}

183DEVICE_MAP fvaGetDeviceMapForImg(const DEVICE_MAP& deviceMap, const QString& pathToFile, QString& deviceName)
184{
185 deviceName = fvaGetExifMakeAndModelFromFile(pathToFile);
186
187 DEVICE_MAP result;
188 if (deviceName.isEmpty())
189 return DEVICE_MAP();
190 QString fixedDevName = deviceName.toUpper().trimmed();
191 for (auto it = deviceMap.begin(); it != deviceMap.end(); ++it)
192 {
193 QString name = it.value().linkedName.toUpper();
194 if (name == fixedDevName)
195 result[it.key()] = it.value();
196 }
197
198 deviceName = deviceName.remove(" ");
199 deviceName = deviceName.remove(QChar('\0'));
200 if (!deviceName.isEmpty())
201 {
202 for (auto it = deviceMap.begin(); it != deviceMap.end(); ++it)
203 {
204 if (it.value().linkedName == deviceName.toUpper().trimmed())
205 result[it.key()] = it.value();
206 }
207 }
208 return result;
209}
